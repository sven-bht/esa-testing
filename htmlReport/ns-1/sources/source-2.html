         Node newNode = new Node(data);

             head = newNode;

             Node current = head;

                 if (i == length - 1) {
                     current.next = newNode;
                 }

             }

             if (i == index) {

         }
         return 0;
     }
 
     public void removeFirst(int data) {
         Node current = head;
         Node previous = null;
         for (int i = 0; i < length; i++) {
             if (current.data == data) {
                 length--;
                 if (length == 0) {
                     head = null;
                 } else if (previous == null) {
                     head = current.next;
                 } else {
                     previous.next = current.next;
                 }
                 return;
             }
         previous = current;
         current = current.next;
         }
     }
